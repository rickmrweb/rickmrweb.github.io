Operadores numéricos

console.log(5 + 2); // Suma -> 7
console.log(5 - 2); // Resta -> 3
console.log(5 * 2); // Multiplicación -> 10
//console.log(5 ** 2); // Potencia -> 25
console.log(5 / 2); // División -> 2.5
console.log(5 % 2); // Módulo o residuo -> 1
