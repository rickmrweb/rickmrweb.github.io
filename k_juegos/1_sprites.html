  <head>

    <meta name="viewport"

    <title>Sprite</title>
    <style>
      /* El selector ".sprite"
       * identifica a todos los
       * elementos cuyo atributo
       * "class" contenga la
       * palabra "sprite" */
      .sprite {
        /* La posición del
         * elemento en la ventana
         * se especifica
         * directamente con las
         * propiedades
         *   left (lado izquierdo
         *        del elemento)
         *   right (lado derecho
         *        del elemento)
         *   top   (lado superior
         *        del elemento)
         *   bottom (lado inferior
         *        del elemento) */
        position: fixed;
        /* El tamaño del font es
         * 3 veces el tamaño de
         * letra para el elemento
         * raíz (en este caso
         * html). */
        font-size: 3rem;
      }
    </style>
  </head>
  <body>
    <h1>Sprite</h1>
    <p>

      colocar una figura en

      ventana.
